### Hello World Code

```javascript
var teacher = "Kyle";
var twitterHandle = "getify";
var age = 39;

  console.log(`Hi, I'm ${myName} (aka ${myNickname}),
      and I'm ${myAge} years old.`);
}

whoAmI(teacher, twitterHandle, age);
```
